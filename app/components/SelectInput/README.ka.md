## ­ЪЊќ рЃЏрЃўрЃЏрЃЮрЃ«рЃўрЃџрЃЋрЃљ

SelectInput рЃЎрЃЮрЃЏрЃърЃЮрЃюрЃћрЃюрЃбрЃў рЃБрЃќрЃарЃБрЃюрЃЋрЃћрЃџрЃДрЃЮрЃцрЃА рЃАрЃбрЃўрЃџрЃўрЃќрЃћрЃЉрЃБрЃџ dropdown select рЃЏрЃћрЃюрЃўрЃБрЃА рЃцрЃЮрЃарЃЏрЃўрЃА рЃўрЃюрЃцрЃБрЃЌрЃћрЃЉрЃўрЃАрЃЌрЃЋрЃўрЃА. рЃЏрЃ«рЃљрЃарЃА рЃБрЃГрЃћрЃарЃА React Hook Form рЃўрЃюрЃбрЃћрЃњрЃарЃљрЃфрЃўрЃљрЃА, рЃЏрЃЮрЃарЃњрЃћрЃЉрЃБрЃџ рЃЋрЃљрЃарЃўрЃљрЃюрЃбрЃћрЃЉрЃА, рЃЋрЃљрЃџрЃўрЃЊрЃљрЃфрЃўрЃљрЃА рЃЊрЃљ рЃерЃћрЃфрЃЊрЃЮрЃЏрЃћрЃЉрЃўрЃА рЃЊрЃљрЃЏрЃБрЃерЃљрЃЋрЃћрЃЉрЃљрЃА.

## Рюе рЃцрЃБрЃюрЃЦрЃфрЃўрЃћрЃЉрЃў

- РюЁ рЃЏрЃЮрЃарЃњрЃћрЃЉрЃБрЃџрЃў dropdown рЃЋрЃљрЃарЃўрЃљрЃюрЃбрЃћрЃЉрЃў
- РюЁ React Hook Form рЃўрЃюрЃбрЃћрЃњрЃарЃљрЃфрЃўрЃљ (Controller)
- РюЁ TypeScript рЃЏрЃ«рЃљрЃарЃЊрЃљрЃГрЃћрЃарЃљ
- РюЁ SCSS рЃЏрЃЮрЃЊрЃБрЃџрЃўрЃА рЃАрЃбрЃўрЃџрЃўрЃќрЃљрЃфрЃўрЃљ
- РюЁ рЃерЃћрЃфрЃЊрЃЮрЃЏрЃћрЃЉрЃўрЃА рЃерЃћрЃбрЃДрЃЮрЃЉрЃўрЃюрЃћрЃЉрЃћрЃЉрЃўрЃА рЃЕрЃЋрЃћрЃюрЃћрЃЉрЃљ
- РюЁ Placeholder рЃЏрЃ«рЃљрЃарЃЊрЃљрЃГрЃћрЃарЃљ
- РюЁ рЃ«рЃћрЃџрЃЏрЃўрЃАрЃљрЃгрЃЋрЃЊрЃЮрЃЏрЃў рЃЊрЃљ рЃАрЃћрЃЏрЃљрЃюрЃбрЃўрЃЎрЃБрЃарЃў HTML
- РюЁ Controlled рЃЎрЃЮрЃЏрЃърЃЮрЃюрЃћрЃюрЃбрЃўрЃА рЃърЃљрЃбрЃћрЃарЃюрЃў

## ­ЪћД TypeScript рЃбрЃўрЃърЃћрЃЉрЃў

  value: string;
  label: string;

  options: inputOption[];
  placeholder?: string;
  value?: string;
  onChange?: (value: string) => void;

### React Hook Form-рЃЌрЃљрЃю (Controller)

```tsx
import { Controller, useForm } from "react-hook-form";

function RegistrationForm() {
  const { control, formState: { errors } } = useForm();

  return (
    <Controller
      name="programmingField"
      control={control}
      render={({ field }) => (
        <SelectInput
          options={PROGRAMMING_FIELD_OPTIONS}
          placeholder="рЃљрЃўрЃарЃЕрЃўрЃћрЃЌ рЃърЃарЃЮрЃњрЃарЃљрЃЏрЃўрЃарЃћрЃЉрЃўрЃА рЃЏрЃўрЃЏрЃљрЃарЃЌрЃБрЃџрЃћрЃЉрЃљ"
          value={field.value}
          onChange={field.onChange}
          error={errors.programmingField?.message}
        />
      )}
    />
  );
}
```

## ­Ъје рЃАрЃбрЃарЃБрЃЦрЃбрЃБрЃарЃљ

```tsx
<div className={styles.selectWrapper}>
  <select
    value={value}
    onChange={(e) => onChange?.(e.target.value)}
    className={styles.select}
  >
    <option value="">{placeholder || "рЃљрЃўрЃарЃЕрЃўрЃћрЃЌ..."}</option>
    {options.map((option) => (
      <option key={option.value} value={option.value}>
        {option.label}
      </option>
    ))}
  </select>
  {error && <span className={styles.error}>{error}</span>}
</div>
```

- рЃЋрЃљрЃарЃўрЃљрЃюрЃбрЃћрЃЉрЃў рЃерЃћрЃўрЃФрЃџрЃћрЃЉрЃљ рЃўрЃЏрЃърЃЮрЃарЃбрЃўрЃарЃЊрЃћрЃА рЃфрЃћрЃюрЃбрЃарЃљрЃџрЃўрЃќрЃћрЃЉрЃБрЃџрЃў рЃЎрЃЮрЃюрЃцрЃўрЃњрЃБрЃарЃљрЃфрЃўрЃўрЃА рЃцрЃљрЃўрЃџрЃћрЃЉрЃўрЃЊрЃљрЃю
